1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.app.service.type }}
  export NODE_PORT=$(kubectl get --namespace  -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Values.app.name }})
  export NODE_IP=$(kubectl get nodes --namespace  -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.app.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace  svc -w {{ .Values.app.name }}'
  export SERVICE_IP=$(kubectl get svc --namespace  {{ .Values.app.name }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.app.service.port }}
{{- else if contains "ClusterIP" .Values.app.service.type }}
  export POD_NAME=$(kubectl get pods --namespace  -l "app={{ .Values.app.name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace  $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace  port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check the status of your deployment:
   kubectl get pods -n 

3. View logs:
   kubectl logs -f deployment/{{ .Values.app.name }} -n 

4. Database connection:
   Database is available at: {{ .Values.database.name }}.svc.cluster.local:{{ .Values.database.service.port }}

5. Memcached connection:
   Memcached is available at: {{ .Values.memcached.name }}.svc.cluster.local:{{ .Values.memcached.service.port }}

6. RabbitMQ connection:
   RabbitMQ AMQP is available at: {{ .Values.rabbitmq.name }}.svc.cluster.local:{{ .Values.rabbitmq.service.amqpPort }}
   RabbitMQ Management is available at: {{ .Values.rabbitmq.name }}.svc.cluster.local:{{ .Values.rabbitmq.service.port }}
